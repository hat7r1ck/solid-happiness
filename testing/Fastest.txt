index="sysmgmt_short" sourcetype="wf:snow:naac:db"
| search (device_name=$tok_entity$ OR user=$tok_entity$ OR assigned_to_logon_id=$tok_entity$ OR user_identity=$tok_entity$)
| eval "Computer Name" = upper(device_name), "Last Seen" = _time
| dedup "Computer Name"
| convert timeformat="%Y-%m-%d %H:%M:%S" ctime("Last Seen")
| eval assigned_user_id_raw = assigned_to_logon_id
| table "Last Seen", "Computer Name", asset_class, environment, os, user, assigned_to_logon_id, assigned_to_name, assigned_user_id_raw
| rename asset_class as "Asset Class",
        environment as "Environment",
        os as "Operating System",
        user as "Last Logged in User",
        assigned_to_logon_id as "Assigned User ID",
        assigned_to_name as "Assigned User Full Name"
| sort -"Last Seen"

index="sysmgmt_short" source="AVGEXT_USER_ACCOUNTS"
| eval _lookup_key = coalesce("$tok_user_from_host$", "$tok_entity$")
| search acct_name=_lookup_key OR elid=_lookup_key OR email_address=_lookup_key
| dedup acct_name

        job_title, lob, sub_lob, manager_name,
        strata_email_1, strata_job_title_1, ad_account_disabled, pwdlastset

        full_name as "Full Name",
        email_address as "Email Address",
        job_title as "Job Title",
